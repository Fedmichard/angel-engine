# All CMake files will always have the same two commands at or near the top: cmake_minimum_required and project:
cmake_minimum_required(VERSION 3.23)

# Name of our project
project(angel-engine)

# C++ standard to use
set(CMAKE_CXX_STANDARD 20)

# The name of our target is angel-engine, and it is an executable. We will add source files to this target later.
add_executable(angel-engine)

# Source files to the angel-engine target
target_sources(angel-engine
	PRIVATE
	src/main.cpp
)
